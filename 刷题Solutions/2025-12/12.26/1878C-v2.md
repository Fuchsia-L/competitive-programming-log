// 1878C 优化版：O(1) 判定
void solve_1878()
{
    long long n, k, x;
    cin >> n >> k >> x;

    // 使用公式直接计算边界，避免 for 循环
    long long min_sum = k * (1 + k) / 2;
    long long max_sum = k * (2 * n - k + 1) / 2;

    if (x >= min_sum && x <= max_sum)
    {
        cout << "YES\n";
    }
    else
    {
        cout << "NO\n";
    }
}
